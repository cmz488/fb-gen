> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Zig as an alternative build system alongside CMake, selectable at `init` time.

**Architecture:** Two independent Tera-template generators in `src/core/` — `CMakeGenerator` (existing, unchanged) and `ZigGenerator` (new, parallel). Both consume the same `Vec<CMakeModule>` + `DependencyGraph` inputs. A `BuildSystem` enum in `ProjectConfig` controls dispatch.

**Tech Stack:** Rust, Tera templates (same as CMake path), Zig (external, on PATH)

### Task 4: Create `ZigGenerator` in `src/core/zig_generator.rs`

**Files:**
- Create: `src/core/zig_generator.rs`

**Context:** Follows the exact same pattern as `CMakeGenerator` — struct holding `config + tera`, `new()` registers templates, `generate()` renders them.

The `build.zig` template is a hard-coded `const` string embedded in the source (same as `ROOT_TEMPLATE` and `MODULE_TEMPLATE` in `generator.rs`).

- [ ] **Step 1: Write the `build.zig` Tera template**

```rust
const ZIG_BUILD_TEMPLATE: &str = r#"// Auto-generated by fb-gen — do not edit manually.
const std = @import("std");

pub fn build(b: *std.Build) void {
    const target = b.resolveTargetQuery(.{
        .cpu_arch = .{{ cpu_arch }},
        .cpu_model = .{ .explicit = &std.Target.{{ cpu_arch }}.cpu.{{ cpu_model }} },
        .os_tag = .{{ os_tag }},
    });
    const optimize = .{{ optimize }};

    // ── Modules ──────────────────────────────────────────────
{% for mod in modules -%}
{% if mod.is_executable -%}
    const {{ mod.name_safe }} = b.addExecutable(.{
        .name = "{{ mod.name }}",
        .target = target,
        .optimize = optimize,
    });
{% else -%}
    const {{ mod.name_safe }} = b.addStaticLibrary(.{
        .name = "{{ mod.name }}",
        .target = target,
        .optimize = optimize,
    });
{% endif -%}
    {{ mod.name_safe }}.addCSourceFiles(.{
        .files = &.{ {% for src in mod.sources -%}"{{ src }}", {% endfor -%} },
        .flags = &.{ {% for flag in mod.compile_flags -%}"{{ flag }}", {% endfor -%} },
    });
    {{ mod.name_safe }}.addIncludePath(.{ .path = "{{ mod.path }}" });
{% for inc in mod.include_dirs -%}
    {{ mod.name_safe }}.addIncludePath(.{ .path = "{{ inc }}" });
{% endfor -%}
{% for def in mod.compile_definitions -%}
    {{ mod.name_safe }}.defineCMacro("{{ def }}");
{% endfor -%}

{% endfor -%}
    // ── Dependencies ──────────────────────────────────────────
{% for edge in dependencies -%}
    {{ edge.from_safe }}.addIncludePath(.{ .path = "{{ edge.to_path }}" });
    {{ edge.from_safe }}.linkLibrary({{ edge.to_safe }});
{% endfor -%}

    // ── Install ──────────────────────────────────────────────
    b.installArtifact({{ root_module }});
}
"#;
```

- [ ] **Step 3: Implement target mapping**

```rust
    /// Map `TargetArch` to a Zig target tuple: `(cpu_arch_str, os_tag_str, default_cpu_model)`.
    fn map_target(arch: &TargetArch) -> (&'static str, &'static str, &'static str) {
        match arch {
            TargetArch::X86_64 => ("x86_64", "linux", "native"),
            TargetArch::X86 => ("x86", "linux", "native"),
            TargetArch::ARM32 => ("arm", "freestanding", "generic"),
            TargetArch::ARM64 => ("aarch64", "freestanding", "generic"),
            TargetArch::NoneEabi => ("thumb", "freestanding", "generic"),
            TargetArch::RISCV64 => ("riscv64", "freestanding", "baseline_rv64"),
            TargetArch::RISCV32 => ("riscv32", "freestanding", "baseline_rv32"),
            TargetArch::Xtensa => ("xtensa", "freestanding", "esp32s3"),
            TargetArch::WASM => ("wasm32", "freestanding", "baseline"),
            TargetArch::Custom(_) => ("x86_64", "linux", "native"),
        }
    }

    /// Resolve the actual CPU model: use `ToolchainConfig.cpu` if set,
    /// otherwise fall back to the architecture default.
    fn resolve_cpu_model(&self, default_model: &str) -> String {
        self.config
            .toolchain
            .as_ref()
            .and_then(|tc| {
                let cpu = &tc.cpu;
                if cpu.is_empty() { None } else { Some(cpu.clone()) }
            })
            .unwrap_or_else(|| default_model.to_string())
    }
```

- [ ] **Step 4: Implement `generate()`**

```rust
    pub fn generate(
        &self,
        modules: &[CMakeModule],
        graph: &DependencyGraph,
        force: bool,
        _user_modules: &[PathBuf],
    ) -> FbGenResult<()> {
        let (cpu_arch, os_tag, default_cpu) = Self::map_target(&self.config.target_arch);
        let cpu_model = self.resolve_cpu_model(default_cpu);
        let optimize = "ReleaseSmall"; // default; could be configurable later

        let root_module = modules
            .iter()
            .find(|m| m.is_root || m.has_main)
            .map(|m| m.name.replace('-', "_"))
            .unwrap_or_else(|| "app".to_string());

        let mut ctx = Context::new();
        ctx.insert("cpu_arch", cpu_arch);
        ctx.insert("cpu_model", &cpu_model);
        ctx.insert("os_tag", os_tag);
        ctx.insert("optimize", optimize);
        ctx.insert("root_module", &root_module);

        // Build module contexts
        let mut mod_ctxs: Vec<serde_json::Value> = Vec::new();
        for m in modules {
            let source_paths: Vec<String> = m.sources.iter()
                .map(|s| s.relative_path.to_string_lossy().to_string())
                .collect();
            mod_ctxs.push(serde_json::json!({
                "name": m.name,
                "name_safe": m.name.replace('-', "_"),
                "path": m.relative_path.to_string_lossy().to_string(),
                "is_executable": m.has_main || m.is_root,
                "sources": source_paths,
                "compile_flags": Vec::<String>::new(),
                "include_dirs": m.include_dirs.iter()
                    .map(|d| d.to_string_lossy().to_string())
                    .collect::<Vec<_>>(),
                "compile_definitions": &m.compile_definitions,
            }));
        }
        ctx.insert("modules", &mod_ctxs);

        // Build dependency edges
        let mut dep_ctxs: Vec<serde_json::Value> = Vec::new();
        for m in modules {
            let from_safe = m.name.replace('-', "_");
            if let Some(deps) = crate::models::dependency::DependencyGraph::get_dependencies_safe(
                graph, &m.name,
            ) {
                for (dep_name, _dep_type) in &deps {
                    // Find the target module to get its path for include dir
                    let to_path = modules.iter()
                        .find(|modu| &modu.name == dep_name)
                        .map(|modu| modu.relative_path.to_string_lossy().to_string())
                        .unwrap_or_default();
                    dep_ctxs.push(serde_json::json!({
                        "from_safe": from_safe,
                        "to_safe": dep_name.replace('-', "_"),
                        "to_path": to_path,
                    }));
                }
            }
        }
        ctx.insert("dependencies", &dep_ctxs);

        let rendered = self.tera.render("build_zig", &ctx)
            .map_err(FbGenError::Template)?;

        let dest = self.config.root.join("build.zig");
        if let Some(parent) = dest.parent() {
            fs::create_dir_all(parent).map_err(FbGenError::Io)?;
        }
        fs::write(&dest, rendered).map_err(FbGenError::Io)?;

        Ok(())
    }
```

**Note:** `DependencyGraph::get_dependencies_safe` doesn't exist yet. It's a wrapper around `get_dependencies()` that returns `Option<Vec<...>>` instead of an empty vec when the module isn't in the graph. Add it in this task:

```rust
// In src/models/dependency.rs, add to impl DependencyGraph:
    /// Like `get_dependencies` but returns None when the module is not in the graph.
    pub fn get_dependencies_safe(&self, name: &str) -> Option<Vec<(String, DependencyType)>> {
        if self.node_map.contains_key(name) {
            Some(self.get_dependencies(name))
        } else {
            None
        }
    }
```

fn generate_build_files(
    config: &ProjectConfig,
    modules: &[CMakeModule],
    graph: &DependencyGraph,
    force: bool,
    user_modules: &[PathBuf],
) -> FbGenResult<()> {
    match config.build_system {
        BuildSystem::CMake => {
            let generator = CMakeGenerator::new(config)?;
            generator.generate(modules, graph, force, user_modules)
        }
        BuildSystem::Zig => {
            let generator = ZigGenerator::new(config)?;
            generator.generate(modules, graph, force, user_modules)
        }
    }
}
